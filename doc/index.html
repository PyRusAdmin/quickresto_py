Авторизация

API использует HTTP Basic Auth — логин и пароль от аккаунта QuickResto, закодированные в Base64.

🔑 Как это работает
Что Значение
Метод HTTP Basic Auth
Заголовок Authorization: Basic <base64(login:password)>
Content-Type application/json обязателен
Connection keep-alive обязателен
🐍 Python — базовая сессия
import requests

# Настройте один раз — используйте везде
session = requests.Session()
session.auth = ('your_login', 'your_password')
session.headers.update({
    'Content-Type': 'application/json',
    'Connection': 'keep-alive',
})

BASE_URL = 'https://demo.quickresto.ru/platform/online'
# Замените 'demo' на имя вашего облака!
ℹ️ Имя облака (layer_name) — это поддомен вашего QuickResto. Например, если вы заходите на myfood.quickresto.ru, то layer_name = myfood.
Структура URL

Все эндпоинты строятся по единому принципу: действие + модуль + класс.

Шаблон
https://{layer_name}.quickresto.ru/platform/online/api/{action}
    ?moduleName={module}
    &className={class}
    &objectId={id}   ← только для /read
Действия (action)
action HTTP Что делает
/api/read GET Получить один объект по ID
/api/list GET Получить список объектов
/api/create POST Создать объект
/api/update POST Изменить объект
/api/remove POST Удалить объект
Пример — клиенты CRM
# module  = crm.customer
# class   = ru.edgex.quickresto.modules.crm.customer.CrmCustomer

GET /api/list
  ?moduleName=crm.customer
  &className=ru.edgex.quickresto.modules.crm.customer.CrmCustomer
read — получить объект

Возвращает один объект со всеми вложенными подобъектами. Нужен objectId.

GET /api/read
Параметр Тип Где Описание
moduleName string query Имя модуля *
className string query Полное имя класса *
objectId integer query ID объекта *
🐍 Пример — получить клиента по ID
def get_customer(customer_id: int) -> dict:
    params = {
        'moduleName': 'crm.customer',
        'className': 'ru.edgex.quickresto.modules.crm.customer.CrmCustomer',
        'objectId': customer_id,
    }
    r = session.get(f'{BASE_URL}/api/read', params=params)
    r.raise_for_status()
    return r.json()

customer = get_customer(42)
print(customer['name'])
list — список объектов

Возвращает массив объектов. Поддерживает фильтрацию, сортировку и пагинацию. Параметры limit/offset передаются в теле запроса (JSON body), остальное — в query.

GET /api/list
Параметр Тип Где Описание
moduleName string query Имя модуля *
className string query Полное имя класса *
limit integer body Макс. кол-во записей необяз.
offset integer body Смещение (пагинация) необяз.
filters array query Фильтры необяз.
sortFields array query Поля сортировки необяз.
sortOrders array query Порядок: asc/desc необяз.
🐍 Пример — список всех клиентов
def list_customers(limit=100, offset=0) -> list:
    params = {
        'moduleName': 'crm.customer',
        'className': 'ru.edgex.quickresto.modules.crm.customer.CrmCustomer',
    }
    body = {'limit': limit, 'offset': offset}
    r = session.get(f'{BASE_URL}/api/list', params=params, json=body)
    r.raise_for_status()
    return r.json()

customers = list_customers(limit=50)
print(f'Получено: {len(customers)} клиентов')
🐍 Пример — список с фильтром и сортировкой
import json

def list_dishes_filtered():
    filters = [
        {'field': 'deleted', 'operation': 'eq', 'value': 'false'},
    ]
    params = {
        'moduleName': 'dish.dish',
        'className': 'ru.edgex.quickresto.modules.dish.Dish',
        'filters': json.dumps(filters),  # ← сериализуем в JSON-строку!
        'sortFields': ['name'],
        'sortOrders': ['asc'],
    }
    body = {'limit': 200}
    r = session.get(f'{BASE_URL}/api/list', params=params, json=body)
    r.raise_for_status()
    return r.json()
create — создать объект

Создаёт новый объект. Данные передаются в теле запроса в формате JSON.

POST /api/create
Параметр Тип Где Описание
moduleName string query Имя модуля *
className string query Полное имя класса *
тело объекта object body Поля нового объекта *
🐍 Пример — создать клиента
def create_customer(name: str, phone: str) -> dict:
    params = {
        'moduleName': 'crm.customer',
        'className': 'ru.edgex.quickresto.modules.crm.customer.CrmCustomer',
    }
    body = {
        'name': name,
        'phone': phone,
    }
    r = session.post(f'{BASE_URL}/api/create', params=params, json=body)
    r.raise_for_status()
    return r.json()  # вернёт созданный объект с id

new_customer = create_customer('Иван Петров', '+79991234567')
print('Создан с ID:', new_customer['id'])

remove — удалить объект

Удаляет объект по ID. В теле передаётся только id.

POST /api/remove
Параметр Тип Где Описание
moduleName string query Имя модуля *

🐍 Пример — удалить объект
def remove_object(module: str, cls: str, obj_id: int) -> dict:
    params = {'moduleName': module, 'className': cls}
    r = session.post(
        f'{BASE_URL}/api/remove',
        params=params,

Фильтрация

Параметр filters — массив объектов. Каждый объект: field + operation + value. Передаётся в query как JSON-строка.

Операции фильтрации
eq
равно
neq
не равно
gt
больше
gte
больше или равно
lt
меньше
lte
меньше или равно
like
содержит подстроку
contains
содержит элемент
range
диапазон дат
🐍 Примеры фильтров
import json

# 1. По точному значению
filters = [{'field': 'deleted', 'operation': 'eq', 'value': 'false'}]

# 2. Поиск по имени (LIKE)
filters = [{'field': 'name', 'operation': 'like', 'value': 'Иван'}]

# 3. Диапазон дат (unix timestamp)
filters = [{
    'field': 'createTime',
    'operation': 'range',
    'value': {
        'since': '1700000000',
        'till': '1710000000',
    }
}]

# 4. Несколько фильтров сразу
filters = [
    {'field': 'deleted', 'operation': 'eq', 'value': 'false'},
    {'field': 'name', 'operation': 'like', 'value': 'пицц'},
]

# Передача в запрос:
params['filters'] = json.dumps(filters)
Сортировка

Параметры sortFields и sortOrders передаются в query. Количество элементов в обоих массивах должно совпадать.

🐍 Примеры сортировки
# По одному полю
params['sortFields'] = ['name']
params['sortOrders'] = ['asc']

# По нескольким полям
params['sortFields'] = ['groupId', 'name']
params['sortOrders'] = ['asc', 'desc']

# Внимание: requests передаст список как повторяющиеся параметры
# ?sortFields=groupId&sortFields=name — это ОК для QuickResto
⚠️ Не все поля поддерживают сортировку — это указано в документации API. При ошибке попробуйте убрать сортировку.
Пагинация

Используйте limit и offset в теле запроса для постраничной загрузки.

🐍 Загрузка всех объектов постранично
def fetch_all(module: str, cls: str, page_size=100) -> list:
    result = []
    offset = 0
    params = {'moduleName': module, 'className': cls}

    while True:
        body = {'limit': page_size, 'offset': offset}
        r = session.get(f'{BASE_URL}/api/list', params=params, json=body)
        r.raise_for_status()
        page = r.json()

        if not page:
            break

        result.extend(page)
        offset += page_size

        if len(page) < page_size:
            break  # последняя страница

    return result

all_dishes = fetch_all('dish.dish', 'ru.edgex.quickresto.modules.dish.Dish')
print(f'Всего блюд: {len(all_dishes)}')
filterCustomers — поиск клиента

Специализированный эндпоинт поиска клиентов по имени, телефону, email или любой строке. Не требует moduleName и className — работает только с клиентами CRM.

ℹ️ URL отличается от стандартного — здесь нет /api/. Путь: /bonuses/filterCustomers
POST /bonuses/filterCustomers
Параметр Тип Где Описание
search string body Строка поиска (имя, телефон, email) *
typeList array body Всегда ["customer"] *
limit integer body Макс. кол-во результатов необяз.
offset integer body Смещение необяз.
🐍 Пример — найти клиента по телефону
def find_customer(search: str) -> list:
    """Поиск клиента по телефону, имени или email"""
    url = f'{BASE_URL}/bonuses/filterCustomers'
    # Обратите внимание: нет params с moduleName/className!
    payload = {
        'search': search,
        'typeList': ['customer'],
        'limit': 10,
        'offset': 0,
    }
    r = session.post(url, json=payload)
    r.raise_for_status()
    return r.json()

# Можно искать по-разному:
find_customer('+79991234567')   # по телефону
find_customer('Иван Петров')     # по имени
find_customer('ivan@mail.ru')    # по email
Сравнение с /api/list
/bonuses/filterCustomers /api/list (crm.customer)
HTTP метод POST GET
Поиск по строке ✓ есть ✗ нет
moduleName/className не нужны обязательны
Фильтры по полям ✗ нет ✓ есть
Когда использовать Поиск по строке Получить всех / фильтр по полю
putCustomer — редактировать клиента

Ещё один специализированный эндпоинт CRM для обновления данных клиента. Альтернатива /api/update для модуля клиентов.

⚠️ URL: /bonuses/putCustomer — без /api/, метод POST, данные в body.
POST /bonuses/putCustomer
Параметр Тип Где Описание
id integer body ID клиента *
поля клиента object body Поля для изменения
🐍 Пример
def put_customer(customer_id: int, **fields) -> dict:
    """Обновить данные клиента через bonuses API"""
    url = f'{BASE_URL}/bonuses/putCustomer'
    payload = {'id': customer_id, **fields}
    r = session.post(url, json=payload)
    r.raise_for_status()
    return r.json()

put_customer(42, email='new@mail.ru', firstName='Иван')
